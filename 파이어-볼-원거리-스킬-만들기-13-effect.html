파이어 볼 원거리 스킬 만들기 13 - Effect

Target은 항상 캐릭터 자신이며 모든 캐릭터는 EffectComponent를 부착한다.

Source 상대 캐릭터이거나 자기 자신이 될수 있다.

13 - 1. Effect 정책

filename : EffectPolicy.cs

Effect 효과에 대한 구체적인 정책

  1. IEffectPolicy 인터페이스

    Apply(효과적용) / Revert(효과 제거) 이 함께 기술 되어야 한다.

    public interface IEffectPolicy { void Apply(Creature target, Creature source, EffectData effectData); void Revert(Creature target, EffectData effectData); /// 효과를 업데이트하는 메서드(독뎀등). 매 프레임 호출됩니다. void UpdateEffect(Creature target, Creature source, EffectData effectData, float deltaTime); }

  2. DamageEffectPolicy

    파이어볼은 Damage정책만 있으면 된다.

    public class DamageEffectPolicy : IEffectPolicy { public void Apply(Creature target, Creature source, EffectData effectData) { if (target == null || source == null || effectData == null) return; float damageValue = source.StatComp.GetTotalStat(Define.EStatType.Attack) * effectData.DamageMultiplier; if (effectData.DamageMultiplier > 0) target.OnDamaged(source,damageValue); } public void Revert(Creature target, EffectData effectData) { // 피해 효과는 즉시 발생하며, 되돌릴 때는 따로 조치가 필요하지 않습니다. } public void UpdateEffect(Creature target, Creature source, EffectData effectData, float deltaTime) { } }

  3. StatEffectPolicyBase : Stat 관련 효과는 비슷한 로직이 있으므로 추상클래스 적용

    public abstract class StatEffectPolicyBase : IEffectPolicy { public virtual void Apply(Creature target, Creature source, EffectData effectData) { if (target == null || effectData == null) return; foreach (StatValuePair pair in effectData.StatValues) { ApplyStat(target, pair); } } public virtual void Revert(Creature target, EffectData effectData) { if (target == null || effectData == null) return; foreach (StatValuePair pair in effectData.StatValues) { RevertStat(target, pair); } } public void UpdateEffect(Creature target, Creature source, EffectData effectData, float deltaTime) { } protected abstract void ApplyStat(Creature target, StatValuePair pair); protected abstract void RevertStat(Creature target, StatValuePair pair); }

  4. BuffStatEffectPolicy : StatEffectPolicyBase

    public class BuffStatEffectPolicy : StatEffectPolicyBase { protected override void ApplyStat(Creature target, StatValuePair pair) { float value = pair.AddValue; float prevValue = target.StatComp.GetTotalStat(pair.StatType); float finalValue = prevValue + value; target.StatComp.SetTotalStat(pair.StatType, finalValue); } protected override void RevertStat(Creature target, StatValuePair pair) { float value = pair.AddValue; float prevValue = target.StatComp.GetTotalStat(pair.StatType); float finalValue = prevValue - value; target.StatComp.SetTotalStat(pair.StatType, finalValue); } public void UpdateEffect(Creature target, Creature source, EffectData effectData, float deltaTime) { }

  5. HealEffectPolicy : StatEffectPolicyBase

    public class HealEffectPolicy : StatEffectPolicyBase { protected override void ApplyStat(Creature target, StatValuePair pair) { target.AddStat(pair.StatType, (int)pair.AddValue, Define.EFontType.Heal); } protected override void RevertStat(Creature target, StatValuePair pair) { // Heal 효과는 되돌릴 필요가 없으므로 빈 구현 } public override void Revert(Creature target, EffectData effectData) { // Heal 효과는 되돌릴 필요가 없으므로 빈 구현 } }

  6. 독뎀등 틱단위 데미지 정책

    DamageOverTimeEffectPolicy : IEffectPolicy

    public class DamageOverTimeEffectPolicy : IEffectPolicy { // 각 EffectId별로 누적 시간을 추적 private Dictionary<int, float> _effectAccumulatedTime = new Dictionary<int, float>(); public void Apply(Creature target, Creature source, EffectData effectData) { if (target == null || effectData == null) return; // 초기 누적 시간 설정 _effectAccumulatedTime[effectData.TemplateId] = 0f; } public void Revert(Creature target, EffectData effectData) { if (target == null || effectData == null) return; // 누적 시간 제거 _effectAccumulatedTime.Remove(effectData.TemplateId); } public void UpdateEffect(Creature target, Creature source, EffectData effectData, float deltaTime) { if (effectData == null) return; if (_effectAccumulatedTime.ContainsKey(effectData.TemplateId) == false) _effectAccumulatedTime[effectData.TemplateId] = 0f; _effectAccumulatedTime[effectData.TemplateId] += deltaTime; while (_effectAccumulatedTime[effectData.TemplateId] >= effectData.TickInterval) { // 주기적으로 피해를 입힙니다. target.OnDamaged(source, effectData.DamagePerTick); _effectAccumulatedTime[effectData.TemplateId] -= effectData.TickInterval; } } }

13 - 2. Effect

filename : Effect.cs

fireball Effect는 DamagePolicy정책을 가진다.

DamagePolicy정책을 가지는 Effect들은 fireball, 매직미사일, 참격등이 있다.

  1. 구성 필수 변수들

    Target : 효과를 적용 받는 대상

    Source : 효과를 시전한 객체

    public int EffectId { get; private set; } public EffectData EffectData { get; private set; } public Define.EEffectType EffectType { get { return EffectData.EffectType; } } public Creature Target { get; private set; } public Creature Source { get; private set; } public IEffectPolicy Policy { get; private set; } public long DespawnTick { get; private set; }

  2. Effect 생성자

    public Effect(int effectId, int templateId, Creature target, Creature source, IEffectPolicy policy) { EffectId = effectId; if (!Managers.Managers.Data.EffectDataDic.TryGetValue(templateId, out EffectData effectData) || effectData == null) throw new ArgumentException($"Invalid Effect Template ID: {templateId}"); EffectData = effectData; Target = target; Source = source; Policy = policy; if (effectData.DurationPolicy == Define.EDurationPolicy.Duration) DespawnTick = Utils.Utils.TickCount + (long)(1000 * effectData.Duration); }

  3. Effect 시전

    Apply : 효과를 적용

    Revert : 효과를 제거

    public virtual void Update() { // 여기서 필요한 경우 효과별 업데이트를 구현 } public virtual void Apply() { Policy?.Apply(Target, Source, EffectData); } public virtual void Revert() { Policy?.Revert(Target, EffectData); }

13 - 3. EffectComponent

filename : EffectComponent.cs

  1. 초기화

    _effects : 다양한 effect 저장

    Policies : 이펙트 정책들을 미리 저장

    public class EffectComponent : MonoBehaviour { private static int _effectIdGenerator = 1; private static int GenerateEffectId() => _effectIdGenerator++; public Creature Target { get; private set; } private Dictionary<int,Effect.Effect> _effects = new Dictionary<int,Effect.Effect>(); public static readonly Dictionary<Define.EEffectType, IEffectPolicy> Policies = new Dictionary<Define.EEffectType, IEffectPolicy>() { { Define.EEffectType.Damage, new DamageEffectPolicy() }, { Define.EEffectType.Heal, new HealEffectPolicy() }, { Define.EEffectType.BuffStat, new BuffStatEffectPolicy() }, { Define.EEffectType.BuffStun, new BuffStunPolicy() }, { Define.EEffectType.Poison, new DamageOverTimeEffectPolicy() }, // 필요한 대로 새 정책을 추가 }; //public void SetInfo(Creature owner) //{ // Target = owner; //} }

  2. 외부에서 Effect 생성,사용 준비

    public static Effect.Effect SpawnEffect(int effectId, int templateId, Creature target, Creature source, IEffectPolicy policy) { Effect.Effect effect = new Effect.Effect(effectId, templateId, target, source, policy); return effect; } public void ApplyEffect(int templateId, Creature source) { if (Managers.Managers.Data.EffectDataDic.TryGetValue(templateId, out EffectData effectData) == false) return; ApplyEffect(effectData, source); } public void ApplyEffect(EffectData effectData, Creature source) { if (effectData == null) return; switch (effectData.DurationPolicy) { case Define.EDurationPolicy.Instant: ApplyImmediateEffect(effectData, source); break; case Define.EDurationPolicy.Duration: ApplyDurationEffect(effectData, source); break; case Define.EDurationPolicy.Infinite: ApplyInfiniteEffect(effectData, source); break; } }

  3. Effect와 정책에 맞게 적용

    void ApplyImmediateEffect(EffectData effectData, Creature source) { if (effectData == null) return; if (Policies.TryGetValue(effectData.EffectType, out IEffectPolicy policy) == false) return; // 1. 이펙트 적용. policy.Apply(Target, source, effectData); } void ApplyDurationEffect(EffectData effectData, Creature source) { if (effectData == null) return; if (effectData.Duration == 0) return; if (Policies.TryGetValue(effectData.EffectType, out IEffectPolicy policy) == false) return; // 1. 이펙트 생성. int effectId = GenerateEffectId(); Effect.Effect effect = SpawnEffect(effectId, effectData.TemplateId, Target, source, policy); _effects.Add(effectId, effect); // 2. 이펙트 적용. effect.Apply(); // 4. 이펙트 소멸 예약. StartCoroutine(CoRemoveEffect(effect,effectData)); } IEnumerator CoRemoveEffect(Effect.Effect effect,EffectData effectData) { yield return new WaitForSeconds(effectData.Duration); RemoveEffect(effect); } void ApplyInfiniteEffect(EffectData effectData, Creature caster) { if (effectData == null) return; if (Policies.TryGetValue(effectData.EffectType, out IEffectPolicy policy) == false) return; // 1. 이펙트 생성. int effectId = GenerateEffectId(); Effect.Effect effect = SpawnEffect(effectId, effectData.TemplateId, Target, caster, policy); _effects.Add(effectId, effect); // 2. 이펙트 적용. effect.Apply(); }

  4. Effect 효과 제거

    public void RemoveItemEffect(int templateId, bool send = true) { Effect.Effect effect = _effects.Values.FirstOrDefault(e => e.EffectData.TemplateId == templateId); if (effect != null) RemoveEffect(effect); } public void RemoveEffect(int effectId) { if (_effects.TryGetValue(effectId, out Effect.Effect effect)) RemoveEffect(effect); } public void RemoveEffect(Effect.Effect effect) { if (effect == null) return; _effects.Remove(effect.EffectId); effect.Revert(); }

  5. 정리 Helper

    public void Clear() { foreach (Effect.Effect effect in _effects.Values.ToList()) RemoveEffect(effect); _effects.Clear(); } /// 소유자에게 무한 기간이 있는 효과를 제외한 모든 효과를 제거합니다. public void ClearExceptInfinity() { foreach (Effect.Effect effect in _effects.Values.ToList()) { if (effect.EffectData.DurationPolicy != Define.EDurationPolicy.Infinite) RemoveEffect(effect); } }

  6. Update : 프레임 틱 효과

    private void Update() { float deltaTime = Time.deltaTime; foreach (var effect in _effects.Values.ToList()) { if (effect.EffectData.DurationPolicy == Define.EDurationPolicy.Duration) { // Policy의 UpdateEffect 호출 effect.Policy?.UpdateEffect(Target, effect.Source, effect.EffectData, deltaTime); } // Infinite 정책도 필요하다면 여기에 추가 if (effect.EffectData.DurationPolicy == Define.EDurationPolicy.Infinite) { // 예를 들어, 지속적으로 스탯을 증가시키는 등의 동작 effect.Policy?.UpdateEffect(Target, effect.Source, effect.EffectData, deltaTime); } } }
