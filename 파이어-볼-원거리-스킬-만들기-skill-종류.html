파이어 볼 원거리 스킬 만들기 15 - Skill 종류

Skill 구현은 검증과 구체적 사용이 기술

15 - 1 Normal Skill

filename : NormalSkill.cs

  • 독 데미지도 추가 되어 있다.

    public class NormalSkill :Skill { public NormalSkill(int templateId, Creature owner) : base(templateId, owner) { } public override void UseSkill(Creature target) { var (failReason, validTarget) = CanUseSkill(target); if (failReason != Define.ECanUseSkillFailReason.None || validTarget == null) return; // 독 스킬 사용 { // MP 소모 if (SkillData.Cost > 0) Owner.AddStat(Define.EStatType.Mp, -SkillData.Cost, Define.EFontType.Cost); // 스킬 애니메이션 등 처리 (필요 시 추가) // 효과 적용 Owner.EffectComp.ApplyEffect(SkillData.EffectDataId, source: Owner); // 쿨타임 업데이트 UpdateCoolTimeInNextUseTick(); } } public override (Define.ECanUseSkillFailReason,Creature) CanUseSkill(Creature target) { var (failReason, validTarget) = base.CanUseSkill(target); if (failReason != Define.ECanUseSkillFailReason.None) return (failReason,validTarget); if (SkillData.EffectData == null) return (Define.ECanUseSkillFailReason.InvalidData, validTarget); return (Define.ECanUseSkillFailReason.None,validTarget); } }

15 - 2 ProjectileSkill

filename : ProjectileSkill.cs

  • public class ProjectileSkill :Skill { public ProjectileSkill(int templateId, Creature owner) : base(templateId, owner) { } public override void UseSkill(Creature target) { var (failReason, validTarget) = CanUseSkill(target); if (failReason != Define.ECanUseSkillFailReason.None || validTarget == null) return; // 투사체 시전 if (SkillData.Cost > 0) Owner.AddStat(Define.EStatType.Mp, -SkillData.Cost, Define.EFontType.Cost); Projectile projectile = Managers.Managers.Object.Spawn<Projectile>(SkillData.ProjectileData.TemplateId); if (projectile != null) { projectile.Init(Owner, this, validTarget); UpdateCoolTimeInNextUseTick(); // 쿨타임 업데이트 } } public override (Define.ECanUseSkillFailReason, Creature) CanUseSkill(Creature target) { var (baseFailReason, validTarget) = base.CanUseSkill(target); if (baseFailReason != Define.ECanUseSkillFailReason.None) return (baseFailReason,validTarget); if (SkillData.ProjectileData == null) return (Define.ECanUseSkillFailReason.InvalidData,validTarget); return (Define.ECanUseSkillFailReason.None,validTarget); } }
