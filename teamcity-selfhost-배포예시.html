    3.bulid step

    Advanced options 선택하면 작업 폴더 설정은 기준경로가 된다.(bin)

    1. Command line을 선택한다.

    2. 결과물은 기본폴더에 있지만 효율적인 작업을 위해 다른 폴더에 결과물을 만든다.

      #!/bin/bash set -e DEST_DIR="result" ZIP_PATTERN="*.zip" if [ -d "$DEST_DIR" ]; then echo "디렉토리 '$DEST_DIR'가 존재합니다. 기존 파일을 삭제합니다." rm -rf "${DEST_DIR:?}/"* else echo "디렉토리 '$DEST_DIR'가 존재하지 않습니다. 생성합니다." mkdir "$DEST_DIR" fi echo "압축을 해제합니다: $ZIP_PATTERN -> $DEST_DIR" unzip -O CP949 $ZIP_PATTERN -d $DEST_DIR echo "압축 해제가 완료되었습니다."

    3.bulid step

    1. Command line을 선택한다.

    2. 작업 경로 bin/

    3. 결과물은 기본폴더에 있지만 효율적인 작업을 위해 다른 폴더에 결과물을 만든다.

      #!/bin/bash set -e # 디버깅을 위해 실행 과정을 출력 set -x # 환경 변수에서 GitHub 토큰 가져오기 GIT_TOKEN=%env.GIT_TOKEN% GIT_USER=bigcat0815 GIT_EMAIL=bigcat0815@gmail.com GIT_REPO=https://$GIT_USER:$GIT_TOKEN@github.com/bigcat0815/yesbigcat.github.io.git TARGET_BRANCH=main # Git 설정 git config --global user.name "$GIT_USER" git config --global user.email "$GIT_EMAIL" # 현재 디렉토리 확인 echo "현재 디렉토리: $(pwd)" # 'site' 디렉토리가 존재하면 삭제 if [ -d "site" ]; then echo "'site' 디렉토리가 이미 존재합니다. 삭제합니다." rm -rf site fi # main 브랜치 클론 echo "GitHub 저장소를 '$TARGET_BRANCH' 브랜치로 클론합니다." git clone --branch $TARGET_BRANCH --single-branch $GIT_REPO site || { echo "Git clone failed"; exit 1; } cd site # 클론된 저장소에 파일이 있는지 확인 if [ "$(ls -A)" ]; then echo "클론된 저장소에 파일이 존재합니다." else echo "클론된 저장소가 비어있습니다. 초기 커밋을 확인하세요." exit 1 fi # 기존 파일 삭제 echo "기존 파일을 삭제합니다." git rm -rf . || echo "삭제할 파일이 없습니다." # 압축 해제된 파일 복사 echo "압축 해제된 파일을 복사합니다." cp -R ../result/* . # 변경 사항 추가 echo "변경 사항을 Git에 추가합니다." git add . # 변경 사항이 있는지 확인 if ! git diff --cached --exit-code > /dev/null; then echo "변경 사항이 있습니다. 커밋하고 푸시합니다." git commit -m "Deploy to GitHub Pages via TeamCity build" git push origin $TARGET_BRANCH echo "배포가 성공적으로 완료되었습니다." else echo "변경 사항이 없습니다. 배포할 내용이 없습니다." fi
