14 Skill 구조

filename : Skill.cs

  1. 초기화

    public abstract class Skill { public int TemplateId { get; set; } protected Creature Owner { get; private set; } public SkillData SkillData { get; set; } // 쿨타임 관리 private long NextUseTick { get; set; } protected Skill(int templateId, Creature owner) { TemplateId = templateId; Owner = owner; if (!Managers.Managers.Data.SkillDataDic.TryGetValue(TemplateId, out SkillData skillData) || skillData == null) { throw new ArgumentException($"Invalid Skill Template ID: {TemplateId}"); } SkillData = skillData; } }

  2. 스킬 사용

    public abstract void UseSkill(Creature target); // 타겟 검증 및 스킬 사용 가능 여부를 판단하고, 유효한 타겟을 반환 public virtual (Define.ECanUseSkillFailReason, Creature) CanUseSkill(Creature target) { if (!IsCoolTimeReady()) return (Define.ECanUseSkillFailReason.CoolTime, null); if (Owner.IsStunned || Owner.ObjectState == Define.EObjectState.Dead) return (Define.ECanUseSkillFailReason.InvalidOwnerState, null); Creature validTarget = ValidateSkillTarget(target); if (validTarget == null) return (Define.ECanUseSkillFailReason.InvalidTarget, null); int dist = Owner.GetDistance(validTarget); if (dist > SkillData.CastingRange) return (Define.ECanUseSkillFailReason.SkillRange, null); if (Owner.StatComp.Mp < SkillData.Cost) return (Define.ECanUseSkillFailReason.SkillCost, null); return (Define.ECanUseSkillFailReason.None, validTarget); } public int GetSkillRange(Creature target) { if (target == null) return 0; return SkillData.CastingRange; } public List<Creature> GatherSkillEffectTargets(Creature target) { List<Creature> targets = new List<Creature>(); if (SkillData.MaxTargetCount == 1) { if (IsValidTargetFriendType(target)) targets.Add(target); } else { bool isSelfTarget = SkillData.TargetType == Define.ETargetType.Self; Vector3 pivot = isSelfTarget ? Owner.transform.position : target.transform.position; targets = Managers.Managers.Object.FindCreatures(pivot, (c) => { if (!IsValidTargetFriendType(c)) return false; int dist = isSelfTarget ? Owner.GetDistance(c) : target.GetDistance(c); return dist <= SkillData.GatherTargetRange; }); } return targets; }

  3. 스킬 검증

    private Creature ValidateSkillTarget(Creature target) { if (SkillData.TargetType == Define.ETargetType.Self) return Owner; if (target == null) return null; if (!IsValidUseSkillTargetType(target)) return null; if (!IsValidTargetFriendType(target)) return null; return target; } private bool IsValidUseSkillTargetType(Creature target) { switch (SkillData.TargetType) { case Define.ETargetType.Self: return Owner == target; case Define.ETargetType.Other: return Owner != target; default: return true; } } private bool IsValidTargetFriendType(Creature target) { switch (SkillData.TargetFriendType) { case Define.ETargetFriendType.Friend: return Owner.IsFriend(target); case Define.ETargetFriendType.Enemy: return Owner.IsEnemy(target); default: return true; } }

  4. 쿨타임 관리

    public bool IsCoolTimeReady() { return GetRemainingCoolTimeInTicks() = 0; } public long GetRemainingCoolTimeInTicks() { return Math.Max(0, NextUseTick - Utils.Utils.TickCount); } private float GetRemainingCoolTimeInSeconds() { return GetRemainingCoolTimeInTicks() / 1000.0f; } public float GetRemainingCoolTimeInRatio() { return GetRemainingCoolTimeInSeconds() / SkillData.Cooltime; } public void UpdateCoolTimeInNextUseTick() { NextUseTick = Utils.Utils.TickCount + (long)(1000 * SkillData.Cooltime); } public void SetRemainingCoolTimeInNextUseTick(long remainTicks) { NextUseTick = Utils.Utils.TickCount + remainTicks; }
